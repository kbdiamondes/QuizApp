SELECT createquiz_quiz.quizid, createquiz_quiz.questionid,createquiz_quiz.subjectname, createquiz_quiz.question, createquiz_quiz.correctanswer, createquiz_quiz.eqpoints, createquiz_studentanswer.studentanswer 
FROM createquiz_quiz 
JOIN createquiz_studentanswer ON createquiz_quiz.questionid = createquiz_studentanswer.questionid_id  
JOIN createquiz_studentanswer_student ON createquiz_studentanswer_student.id = createquiz_studentanswer.id
WHERE createquiz_studentanswer.id = createquiz_studentanswer_student.id 
AND createquiz_studentanswer_student.student_id = username
